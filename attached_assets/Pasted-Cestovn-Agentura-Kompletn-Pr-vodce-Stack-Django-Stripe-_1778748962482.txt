pythonfrom django.db import models

class Trip(models.Model):
    nazev = models.CharField(max_length=200)
    destinace = models.CharField(max_length=200)
    popis = models.TextField()
    cena = models.DecimalField(max_digits=10, decimal_places=2)   # v CZK
    pocet_dni = models.IntegerField()
    obrazek_url = models.URLField(blank=True)
    volna_mista = models.IntegerField(default=10)
    aktivni = models.BooleanField(default=True)

    def __str__(self):
        return self.nazev

    def zalohova_castka(self):
        """30% záloha zaokrouhlená na celé koruny"""
        return round(self.cena * 30 / 100)


class Rezervace(models.Model):
    STAV_CHOICES = [
        ('ceka', 'Čeká na platbu'),
        ('zalohazaplacena', 'Záloha zaplacena'),
        ('zaplaceno', 'Plně zaplaceno'),
        ('zruseno', 'Zrušeno'),
    ]
    TYP_PLATBY_CHOICES = [
        ('karta', 'Karta / Google Pay / Apple Pay'),
        ('qr', 'QR Platba'),
    ]
    TYP_REZERVACE_CHOICES = [
        ('zaloha', 'Záloha 30 %'),
        ('plna', 'Plná platba'),
    ]

    zajezd = models.ForeignKey(Trip, on_delete=models.CASCADE)
    jmeno = models.CharField(max_length=200)
    email = models.EmailField()
    telefon = models.CharField(max_length=30, blank=True)
    typ_rezervace = models.CharField(max_length=10, choices=TYP_REZERVACE_CHOICES)
    typ_platby = models.CharField(max_length=10, choices=TYP_PLATBY_CHOICES)
    castka = models.DecimalField(max_digits=10, decimal_places=2)
    stav = models.CharField(max_length=20, choices=STAV_CHOICES, default='ceka')
    stripe_session_id = models.CharField(max_length=200, blank=True)
    vytvoreno = models.DateTimeField(auto_now_add=True)

    def __str__(self):
        return f"{self.jmeno} — {self.zajezd.nazev} ({self.get_stav_display()})"

ČÁST 5 — Zpracování rezervace (view)
Přidej do trips/views.py:
pythondef rezervace(request, pk):
    zajezd = get_object_or_404(Trip, pk=pk, aktivni=True)

    if request.method != 'POST':
        return redirect('detail', pk=pk)

    typ_rezervace = request.POST.get('typ_rezervace', 'zaloha')
    typ_platby = request.POST.get('typ_platby', 'karta')
    jmeno = request.POST.get('jmeno', '')
    email = request.POST.get('email', '')
    telefon = request.POST.get('telefon', '')

    # Vypočítej částku
    if typ_rezervace == 'zaloha':
        castka = zajezd.zalohova_castka()
    else:
        castka = zajezd.cena

    # Ulož rezervaci
    rezervace = Rezervace.objects.create(
        zajezd=zajezd,
        jmeno=jmeno,
        email=email,
        telefon=telefon,
        typ_rezervace=typ_rezervace,
        typ_platby=typ_platby,
        castka=castka,
        stav='ceka',
    )

    # Přesměruj na správnou platební metodu
    if typ_platby == 'karta':
        return redirect('stripe_checkout', pk=rezervace.pk)
    else:
        return redirect('qr_platba', pk=rezervace.pk)

def stripe_checkout(request, pk):
    rezervace_obj = get_object_or_404(Rezervace, pk=pk)
    zajezd = rezervace_obj.zajezd
    DOMAIN = request.build_absolute_uri('/').rstrip('/')

    popis = f"{'Záloha 30 %' if rezervace_obj.typ_rezervace == 'zaloha' else 'Plná platba'} — {zajezd.nazev}"

    session = stripe.checkout.Session.create(
        payment_method_types=['card'],   # Google/Apple Pay se přidá automaticky
        line_items=[{
            'price_data': {
                'currency': 'czk',
                'product_data': {
                    'name': zajezd.nazev,
                    'description': popis,
                },
                'unit_amount': int(rezervace_obj.castka * 100),  # haléře
            },
            'quantity': 1,
        }],
        mode='payment',
        success_url=DOMAIN + '/uspech/?session_id={CHECKOUT_SESSION_ID}',
        cancel_url=DOMAIN + f'/zajezd/{zajezd.pk}/',
        customer_email=rezervace_obj.email,
        metadata={
            'rezervace_id': rezervace_obj.pk,
        },
    )

    rezervace_obj.stripe_session_id = session.id
    rezervace_obj.save()

    return redirect(session.url, code=303)


@csrf_exempt
def stripe_webhook(request):
    payload = request.body
    sig_header = request.META.get('HTTP_STRIPE_SIGNATURE')

    try:
        event = stripe.Webhook.construct_event(
            payload, sig_header, settings.STRIPE_WEBHOOK_SECRET
        )
    except Exception:
        return HttpResponse(status=400)

    if event['type'] == 'checkout.session.completed':
        session = event['data']['object']
        rezervace_id = session['metadata'].get('rezervace_id')
        if rezervace_id:
            try:
                r = Rezervace.objects.get(pk=rezervace_id)
                if r.typ_rezervace == 'zaloha':
                    r.stav = 'zalohazaplacena'
                else:
                    r.stav = 'zaplaceno'
                r.save()
            except Rezervace.DoesNotExist:
                pass

    return HttpResponse(status=200)

ČÁST 7 — QR Platba (česká bankovní platba)
Jak to funguje
QR platba používá český standard SPD (Short Payment Descriptor). Zákazník naskenuje QR kód ve své bankovní aplikaci (Česká spořitelna, ČSOB, KB, Raiffeisenbank…) a platba se mu předvyplní. Peníze přijdou přímo na tvůj účet — 0 % provize.
7.1 Instalace knihovny pro QR kód
bashpip install qrcode[pil]
Přidej qrcode[pil] do requirements.txt.
7.2 QR view
Přidej do trips/views.py:
pythonimport qrcode
import io
import base64

# Sem doplň svůj skutečný IBAN (formát: CZ0000000000000000000000)
TVUJ_IBAN = "CZ0000000000000000000000"  # ← ZMĚŇ TOTO
NAZEV_PRIJEMCE = "Kyrgyzstán Zájezdy s.r.o."

def qr_platba(request, pk):
    rezervace_obj = get_object_or_404(Rezervace, pk=pk)
    zajezd = rezervace_obj.zajezd

    # Vygeneruj variabilní symbol (číslo rezervace)
    vs = str(rezervace_obj.pk).zfill(10)

    # SPD řetězec — český standard pro QR platby
    castka_str = f"{rezervace_obj.castka:.2f}"
    zprava = f"Zaloha zajezd {zajezd.nazev[:20]}"

    spd = (
        f"SPD*1.0*"
        f"ACC:{TVUJ_IBAN}*"
        f"AM:{castka_str}*"
        f"CC:CZK*"
        f"MSG:{zprava}*"
        f"X-VS:{vs}"
    )

    # Vygeneruj QR obrázek
    qr = qrcode.QRCode(version=1, box_size=8, border=4)
    qr.add_data(spd)
    qr.make(fit=True)
    img = qr.make_image(fill_color="black", back_color="white")

    # Převeď na base64 pro zobrazení v HTML
    buffer = io.BytesIO()
    img.save(buffer, format='PNG')
    qr_base64 = base64.b64encode(buffer.getvalue()).decode()

    context = {
        'rezervace': rezervace_obj,
        'zajezd': zajezd,
        'iban': TVUJ_IBAN,
        'castka': rezervace_obj.castka,
        'vs': vs,
        'nazev_prijemce': NAZEV_PRIJEMCE,
        'qr_base64': qr_base64,
    }
    return render(request, 'trips/qr_platba.html', context)
